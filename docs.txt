# Variable (Lists) (They are in order)

>>> VAR list = [1, 2, 3]
[1, 2, 3]

>>> list
[1, 2, 3]

>>> APPEND(list, 4)
0

>>> list
[1, 2, 3, 4]

>>> POP(list, 3)
4

>>> list
[1, 2, 3]

>>> EXTEND(list, [4, 5, 6])
0

>>> list
[1, 2, 3, 4, 5, 6]
